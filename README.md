# PRE
1. npm install

# DEVELOP
1. npm start
2. open in browser: http://IP:PORT

# BUILD
1. chmod 740 build.sh
2. npm run build

# BUILD MOBILE
1. chmod 740 build.sh
2. npm run build mobile

# RUN BUILDED
1. npm run dist-start
2. open in browser: http://IP:PORT
