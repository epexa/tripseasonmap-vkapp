# PRE
1. npm install

# BUILD
1. chmod 740 build.sh
2. npm run build

# BUILD MOBILE
1. chmod 740 build.sh
2. npm run build mobile

# DEVELOP
1. npm start
2. open in browser: http://127.0.0.1:8081
